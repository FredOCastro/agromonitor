"""
fetch_data.py
Coleta dados de todas as fontes externas (APIs gratuitas).
Salva tudo em src/data_atual.json para o analyze.py processar.
"""

import json
import os
import requests
from datetime import datetime, date

# ─── Helpers ───────────────────────────────────────────────────────────────────

def get(url, params=None, timeout=15):
    """Faz GET com tratamento de erro básico."""
    try:
        r = requests.get(url, params=params, timeout=timeout)
        r.raise_for_status()
        return r.json()
    except Exception as e:
        print(f"  [ERRO] {url}: {e}")
        return None


# ─── 1. ENSO / ONI — NOAA ──────────────────────────────────────────────────────

def fetch_enso():
    """
    Busca o índice ONI (Oceanic Niño Index) da NOAA.
    Retorna o valor mais recente e a tendência.
    """
    print("Buscando dados ENSO (NOAA)...")
    url = "https://www.cpc.ncep.noaa.gov/data/indices/oni.ascii.txt"
    try:
        r = requests.get(url, timeout=15)
        linhas = r.text.strip().split("\n")
        # Pega as últimas 3 linhas com dados (ignora cabeçalho)
        dados = []
        for linha in linhas:
            partes = linha.split()
            if len(partes) >= 3:
                try:
                    dados.append({
                        "periodo": partes[0],
                        "ano": int(partes[1]),
                        "oni": float(partes[2])
                    })
                except (ValueError, IndexError):
                    continue

        if not dados:
            raise ValueError("Sem dados ONI")

        atual = dados[-1]
        anterior = dados[-2] if len(dados) > 1 else atual
        tendencia = "subindo" if atual["oni"] > anterior["oni"] else "caindo"

        # Classifica o status
        oni = atual["oni"]
        if oni >= 1.5:
            status = "El Niño Forte"
        elif oni >= 0.5:
            status = "El Niño Fraco/Moderado"
        elif oni <= -1.5:
            status = "La Niña Forte"
        elif oni <= -0.5:
            status = "La Niña Fraca/Moderada"
        else:
            status = "Neutro"

        return {
            "oni_atual": oni,
            "oni_anterior": anterior["oni"],
            "periodo": atual["periodo"],
            "status": status,
            "tendencia": tendencia,
            "fonte": "NOAA CPC"
        }
    except Exception as e:
        print(f"  [ERRO] ENSO: {e}")
        # Valor fallback baseado em dados recentes conhecidos
        return {
            "oni_atual": 0.4,
            "oni_anterior": 0.3,
            "periodo": "MAM 2026",
            "status": "Neutro (tendência El Niño)",
            "tendencia": "subindo",
            "fonte": "NOAA CPC (fallback)"
        }


# ─── 2. Clima regional — Open-Meteo (gratuito, sem chave) ─────────────────────

def fetch_clima_regional():
    """
    Busca previsão do tempo para as 4 cidades do Triângulo Mineiro.
    Open-Meteo é 100% gratuito e não precisa de API key.
    """
    print("Buscando clima regional (Open-Meteo)...")
    cidades = {
        "Araxá":    {"lat": -19.593, "lon": -46.942},
        "Ibiá":     {"lat": -19.475, "lon": -46.540},
        "Perdizes":  {"lat": -19.342, "lon": -47.293},
        "Tapira":   {"lat": -19.898, "lon": -46.829},
    }

    resultado = {}
    for cidade, coords in cidades.items():
        data = get(
            "https://api.open-meteo.com/v1/forecast",
            params={
                "latitude": coords["lat"],
                "longitude": coords["lon"],
                "daily": "precipitation_sum,temperature_2m_max,temperature_2m_min,et0_fao_evapotranspiration",
                "timezone": "America/Sao_Paulo",
                "forecast_days": 14,
            }
        )
        if data and "daily" in data:
            precip_7d = sum(p or 0 for p in data["daily"]["precipitation_sum"][:7])
            precip_14d = sum(p or 0 for p in data["daily"]["precipitation_sum"][:14])
            temp_max = max(t for t in data["daily"]["temperature_2m_max"][:7] if t is not None)
            temp_min = min(t for t in data["daily"]["temperature_2m_min"][:7] if t is not None)
            resultado[cidade] = {
                "precip_7d_mm": round(precip_7d, 1),
                "precip_14d_mm": round(precip_14d, 1),
                "temp_max_semana": round(temp_max, 1),
                "temp_min_semana": round(temp_min, 1),
                "alerta_seca": precip_7d < 10,
                "alerta_excesso": precip_7d > 120,
            }
        else:
            resultado[cidade] = {"erro": "sem dados"}

    return resultado


# ─── 3. Câmbio USD/BRL — Banco Central do Brasil ──────────────────────────────

def fetch_cambio():
    """Busca cotação USD/BRL do Bacen (API oficial, gratuita)."""
    print("Buscando câmbio (Bacen)...")
    today = date.today().strftime("%m-%d-%Y")
    url = f"https://olinda.bcb.gov.br/olinda/servico/PTAX/versao/v1/odata/CotacaoDolarDia(dataCotacao=@dataCotacao)?@dataCotacao='{today}'&$format=json"
    data = get(url)
    if data and "value" in data and data["value"]:
        cotacao = data["value"][-1]
        return {
            "usd_brl": round(cotacao["cotacaoVenda"], 4),
            "data": today,
            "fonte": "Banco Central do Brasil"
        }
    # Fallback: tenta dia anterior
    from datetime import timedelta
    yesterday = (date.today() - timedelta(days=1)).strftime("%m-%d-%Y")
    url2 = f"https://olinda.bcb.gov.br/olinda/servico/PTAX/versao/v1/odata/CotacaoDolarDia(dataCotacao=@dataCotacao)?@dataCotacao='{yesterday}'&$format=json"
    data2 = get(url2)
    if data2 and "value" in data2 and data2["value"]:
        cotacao = data2["value"][-1]
        return {
            "usd_brl": round(cotacao["cotacaoVenda"], 4),
            "data": yesterday,
            "fonte": "Banco Central do Brasil"
        }
    return {"usd_brl": 5.95, "data": today, "fonte": "Bacen (fallback)"}


# ─── 4. Preços futuros — Yahoo Finance (sem chave) ────────────────────────────

def fetch_precos_futuros():
    """
    Busca preços futuros de soja e milho via Yahoo Finance.
    ZS=F = Soja CBOT (cents/bushel), ZC=F = Milho CBOT (cents/bushel)
    """
    print("Buscando preços futuros (Yahoo Finance)...")

    tickers = {
        "soja_cbot": "ZS=F",
        "milho_cbot": "ZC=F",
    }

    resultado = {}
    for nome, ticker in tickers.items():
        url = f"https://query1.finance.yahoo.com/v8/finance/chart/{ticker}"
        headers = {"User-Agent": "Mozilla/5.0"}
        try:
            r = requests.get(url, headers=headers, timeout=15)
            data = r.json()
            meta = data["chart"]["result"][0]["meta"]
            preco_cents = meta.get("regularMarketPrice", 0)
            preco_usd_bu = round(preco_cents / 100, 2)  # converte cents para USD
            resultado[nome] = {
                "preco_usd_bushel": preco_usd_bu,
                "ticker": ticker,
                "moeda": "USD/bushel",
            }
        except Exception as e:
            print(f"  [ERRO] Yahoo {ticker}: {e}")
            # Fallbacks baseados em valores recentes
            fallbacks = {"soja_cbot": 10.42, "milho_cbot": 4.58}
            resultado[nome] = {
                "preco_usd_bushel": fallbacks[nome],
                "ticker": ticker,
                "moeda": "USD/bushel",
                "fonte": "fallback"
            }

    # Calcula preço em R$/saca (60kg) usando câmbio atual
    cambio = fetch_cambio()
    usd_brl = cambio["usd_brl"]
    # 1 bushel soja = 27.2kg → 1 saca 60kg = ~2.2 bushels
    if "soja_cbot" in resultado:
        p = resultado["soja_cbot"]["preco_usd_bushel"]
        resultado["soja_cbot"]["preco_brl_saca"] = round(p * 2.2 * usd_brl, 2)
    # 1 bushel milho = 25.4kg → 1 saca 60kg = ~2.36 bushels
    if "milho_cbot" in resultado:
        p = resultado["milho_cbot"]["preco_usd_bushel"]
        resultado["milho_cbot"]["preco_brl_saca"] = round(p * 2.36 * usd_brl, 2)

    resultado["cambio_usd_brl"] = usd_brl
    return resultado

# ─── Main ──────────────────────────────────────────────────────────────────────

def main():
    print("\n" + "="*50)
    print("AGRO MONITOR — Coleta de dados")
    print(f"Data/hora: {datetime.now().strftime('%d/%m/%Y %H:%M')}")
    print("="*50 + "\n")

    dados = {
        "timestamp": datetime.now().isoformat(),
        "data_coleta": datetime.now().strftime("%d/%m/%Y %H:%M"),
        "enso": fetch_enso(),
        "clima": fetch_clima_regional(),
        "precos": fetch_precos_futuros(),
        "safras": fetch_safras_usda(),
    }

    # Salva os dados coletados
    output_path = os.path.join(os.path.dirname(__file__), "data_atual.json")
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(dados, f, ensure_ascii=False, indent=2)

    print(f"\nDados salvos em: {output_path}")
    print(f"ONI atual: {dados['enso']['oni_atual']} ({dados['enso']['status']})")
    print(f"Câmbio: R$ {dados['precos']['cambio_usd_brl']}")
    print(f"Soja: USD {dados['precos']['soja_cbot']['preco_usd_bushel']}/bu = R$ {dados['precos']['soja_cbot'].get('preco_brl_saca','?')}/sc")
    print(f"Milho: USD {dados['precos']['milho_cbot']['preco_usd_bushel']}/bu = R$ {dados['precos']['milho_cbot'].get('preco_brl_saca','?')}/sc")
    print("\nColeta concluída!\n")


if __name__ == "__main__":
    main()
